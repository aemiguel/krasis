fn quicksort<T: Ord + Copy>(arr: &mut [T]) {
    quicksort_recursive(arr, 0, arr.len().saturating_sub(1));
}

fn quicksort_recursive<T: Ord + Copy>(arr: &mut [T], low: usize, high: usize) {
    if low < high {
        let partition_index = partition(arr, low, high);
        if partition_index > 0 {
            quicksort_recursive(arr, low, partition_index.saturating_sub(1));
            quicksort_recursive(arr, partition_index.saturating_add(1), high);
        }
    }
}

fn partition<T: Ord + Copy>(arr: &mut [T], low: usize, high: usize) -> usize {
    let pivot = arr[high];
    let mut i = low.saturating_sub(1);

    for j in low..high {
        if arr[j] <= pivot {
            i = i.saturating_add(1);
            arr.swap(i, j);
        }
    }
    arr.swap(i.saturating_add(1), high);
    i.saturating_add(1)
}
