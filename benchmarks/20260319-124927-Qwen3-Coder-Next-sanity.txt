pub fn quicksort<T: Ord + Clone>(arr: &mut [T]) {
    if arr.len() <= 1 {
        return;
    }

    let pivot = arr[arr.len() / 2].clone();
    let (mut left, mut right) = (0usize, arr.len() - 1);

    loop {
        // Move left index until we find an element >= pivot
        while left < arr.len() && &arr[left] < &pivot {
            left += 1;
        }

        // Move right index until we find an element <= pivot
        while right > 0usize && &arr[right] > &pivot {
            right = if right >= 1 { right - 1 } else { 0usize };
        }

        // If pointers cross, break
        if left >= right {
            break;
        }

        // Swap elements at left and right
        arr.swap(left, right);

        // Move pointers inward
        left += 1;
        right = if right >= 1 { right - 1 } else { 0usize };
